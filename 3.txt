--Attendance
SET SERVEROUTPUT ON;
DECLARE
    RNO NUMBER;
    ATT NUMBER;
    ST VARCHAR2(20);
    
BEGIN
    RNO:= &RNO;
    SELECT ATTENDANCE INTO ATT FROM STUD WHERE ROLL_NO=RNO;
    IF (ATT>=75)THEN
        ST := 'ND';
        DBMS_OUTPUT.PUT_LINE('TERM GRANTED....!');
    ELSE
        ST:= 'D';
        DBMS_OUTPUT.PUT_LINE('TERM NOT GRANTED....!');
    END IF;
    UPDATE STUD SET STATUS=ST WHERE ROLL_NO=RNO;
    
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('ROLL NUMBERNOT FOUND IN TABLE');
END;
